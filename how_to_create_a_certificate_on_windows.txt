1. generate private key
openssl genrsa -out server.key 1024

2. 
openssl req -newkey rsa:4096 -sha256 -nodes -keyout server.key -x509 -days 365 -out server.pem -subj "/C=UA/ST=Chernivtsi/L=Bukovina/O=Computerra/CN=192.168.1.3"
#
Your server.pem has to be used as input for setting the self-signed webhook.


You can inspect the generated certificate with:
openssl x509 -text -noout -in server.pem
